// ðŸ”¹ Convertir texto a audio con ElevenLabs
async function textoAAudio(texto, nombreArchivo) {

    const partes = texto.match(/.{1,900}(\s|$)/g);
    const audioBuffers = [];

    for (const parte of partes) {
      console.log(`ðŸŽ§ Generando audio con ElevenLabs...`);
      const response = await axios.post(
        'https://api.elevenlabs.io/v1/text-to-speech/EiNlNiXeDU1pqqOPrYMO', // tu voice_id
        {
          text: parte.trim(),
          model_id: 'eleven_multilingual_v2',
          voice_settings: {
            stability: 0.65,
            similarity_boost: 0.85,
            style: 0.3,
          },
        },
        {
          headers: {
            'xi-api-key': ELEVEN_API_KEY,
            'Content-Type': 'application/json',
          },
          responseType: 'arraybuffer',
        }
      );

      audioBuffers.push(Buffer.from(response.data));
    }

    fs.writeFileSync(rutaArchivo, Buffer.concat(audioBuffers));
    console.log(`âœ… Archivo de audio creado: ${rutaArchivo}`);
  } catch (error) {
    console.error('âŒ Error al convertir texto a audio (ElevenLabs):', error.message);
    if (error.response) {
      console.error('ðŸ“© Respuesta de la API:', error.response.status, error.response.data);
    }
  }
}

// ðŸ”¹ LIMPIA texto y corta donde termina una oraciÃ³n REAL (.)
function limpiarYArreglarTexto(texto) {
  if (!texto) return "";

  texto = texto
    .replace(/\[\+\d+\schars\]/g, '')  // elimina [+123 chars]
    .replace(/â€¦/g, '')                 // elimina "â€¦"
    .replace(/\[.*?\]/g, '')           // elimina cualquier contenido entre []
    .replace(/\s+/g, ' ')              // limpia espacios dobles
    .trim();

  // Cortar en el Ãºltimo punto real
  const ultimoPunto = texto.lastIndexOf('.');
  if (ultimoPunto !== -1) {
    return texto.substring(0, ultimoPunto + 1);
  }

  return texto;
}


// ðŸ”¹ Obtener noticia completa desde NewsAPI
async function obtenerNoticias() {
  try {
    const url = `https://newsapi.org/v2/everything?q=futbol+(ecuador+OR+europa+OR+uefa+OR+champions)&language=es&sortBy=publishedAt&pageSize=10&apiKey=${API_KEY}`;
    const response = await axios.get(url);
    const noticias = response.data.articles;

    if (!noticias.length) return null;

    const noticia = noticias[Math.floor(Math.random() * noticias.length)];

    const partes = [
      noticia.title || "",
      noticia.description || "",
      noticia.content || ""
    ];

    // Unir todos los textos de la noticia
    let texto = partes.join('. ');

    // Limpiar y arreglar
    texto = limpiarYArreglarTexto(texto);

    return `Flash informativo de Ego Radio Digital. ${texto}`;

  } catch (error) {
    console.error('âŒ Error al obtener noticias:', error.message);
    return null;
  }
}


// ðŸ”¹ Convertir texto a audio usando gTTS

    console.log('ðŸŽ§ Generando audio con gTTS...');

    return new Promise((resolve, reject) => {

        if (err) {
          console.error('âŒ Error en gTTS:', err);
          reject(err);
        } else {
          console.log(`âœ… Audio creado: ${rutaArchivo}`);
          resolve();
        }
      });
    });

  } catch (error) {
    console.error('âŒ Error al convertir texto a audio:', error.message);
  }
}
